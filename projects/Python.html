Functions, Random Number Generators (Lotto Game):

Objective: Write a program that will prompt a user for three numbers where the numbers will be then checked against a computer generated series of three random numbers that are in sorted order. If the user numbers match each number generated by the computer as sorted they win.

View Source Code 

# imports
import random

# global variables
PROGRAM_INFO = '\nWritten by Brandon Sansone\nDate: 6/28/2016\n
NUMBER_OF_PICKS = 3
MIN_PICK = 0
MAX_PICK = 9
WINNINGS = 100
OFFSETT = 4
ATTEMPTS = 0
MAX_ATTEMPTS = 3

# 'checker', compares userNums and winningNums to see if they have won or lost,
# prints the output and returns True/False
def checker(userNums, winningNums):
    if userNums == winningNums:
        print ("\nCongratulations! You Win ${}!".format(WINNINGS),
               "\nYour numbers: ", userNums,
               "\nThe winning lottery numbers were: ", winningNums, "\n")
        return True

    else:
        print ("\nSorry, you lose...",
                "\nYour numbers: ", userNums,
                "\nThe winning lottery numbers were: ", winningNums, "\n")
        return False


# 'get_user_nums', gets user input and puts into a list,
# prints the num of attempts user is on, returns sorted list
def get_user_nums():
    userNums = []
    print ("Attempt #", ATTEMPTS)
    while len(userNums) < NUMBER_OF_PICKS:
        nums = input("Pick a number {} through {}: ".format(MIN_PICK, MAX_PICK))
        try:
            nums = int(nums)
        except:
            print("Sorry your input must be an integer!")
            continue
        if MIN_PICK <= nums <= MAX_PICK:
            if nums not in userNums:
                userNums.append(nums)
            else:
                print("Error! You have already picked this number")
        else:
            print("Error! Your number was not in range")

    return sorted(userNums)


# 'get_winning_nums', creates a sorted list with random nums 
# ranging from 0-9 with a range of 3 values
def get_winning_nums():
    return sorted(random.sample(range(MIN_PICK, MAX_PICK), NUMBER_OF_PICKS))


# 'lottery_menu', creates the text main menu to choose game or exit program
def lottery_menu():
    name = ' '*int(OFFSETT/2) + "LOTTERY MENU"
    dotted = (OFFSETT+len(name))*'-'
    options = ["[Play Pick {}]".format(NUMBER_OF_PICKS),
               "[Exit]"]
    print('{} \n{} \n{}'.format(dotted, name, dotted))
    for i, opt in enumerate(options):
        print(i+1, opt)
    print(dotted)


# 'play_pick_n', this is the game, calls the functions that get you the,
# winningNums, userNums, and checker, does with this within 3 attempts
def play_pick_n():
    global ATTEMPTS
    winningNums = get_winning_nums()

    while ATTEMPTS < MAX_ATTEMPTS:
        ATTEMPTS += 1
        userNums = get_user_nums()
        if checker(userNums, winningNums):
            break


# 'main', calls the other functions, and builds the menu
def main():
    print (PROGRAM_INFO)
    lottery_menu()
    while True:
        choice = input("\nEnter your choice[1-2]: ")
        if choice == '1':
            string = "\n[Play Pick {}]".format(NUMBER_OF_PICKS) + "selected!"
            dotted = '\n'+ len(string) * "-"

            print(dotted,
                  string,
                  dotted)

            play_pick_n()
            break

        elif choice == '2':
            print ("Thanks for playing!\n")
            break

        print("Error! Invalid input. Press any key to continue...\n")

if __name__ == '__main__':
    main()
