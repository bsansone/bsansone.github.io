Functions, Random Number Generators (Lotto Game):

Objective: Write a program that will prompt a user for three numbers where the numbers will be then checked against a computer generated series of three random numbers that are in sorted order. If the user numbers match each number generated by the computer as sorted they win.

View Source Code 

# imports
import random

# global variables
PROGRAM_INFO = '\nWritten by Brandon Sansone\nDate: 6/28/2016\n
NUMBER_OF_PICKS = 3
MIN_PICK = 0
MAX_PICK = 9
WINNINGS = 100
OFFSETT = 4
ATTEMPTS = 0
MAX_ATTEMPTS = 3

# 'checker', compares userNums and winningNums to see if they have won or lost,
# prints the output and returns True/False
def checker(userNums, winningNums):
    if userNums == winningNums:
        print ("\nCongratulations! You Win ${}!".format(WINNINGS),
               "\nYour numbers: ", userNums,
               "\nThe winning lottery numbers were: ", winningNums, "\n")
        return True

    else:
        print ("\nSorry, you lose...",
                "\nYour numbers: ", userNums,
                "\nThe winning lottery numbers were: ", winningNums, "\n")
        return False


# 'get_user_nums', gets user input and puts into a list,
# prints the num of attempts user is on, returns sorted list
def get_user_nums():
    userNums = []
    print ("Attempt #", ATTEMPTS)
    while len(userNums) < NUMBER_OF_PICKS:
        nums = input("Pick a number {} through {}: ".format(MIN_PICK, MAX_PICK))
        try:
            nums = int(nums)
        except:
            print("Sorry your input must be an integer!")
            continue
        if MIN_PICK <= nums <= MAX_PICK:
            if nums not in userNums:
                userNums.append(nums)
            else:
                print("Error! You have already picked this number")
        else:
            print("Error! Your number was not in range")

    return sorted(userNums)


# 'get_winning_nums', creates a sorted list with random nums 
# ranging from 0-9 with a range of 3 values
def get_winning_nums():
    return sorted(random.sample(range(MIN_PICK, MAX_PICK), NUMBER_OF_PICKS))


# 'lottery_menu', creates the text main menu to choose game or exit program
def lottery_menu():
    name = ' '*int(OFFSETT/2) + "LOTTERY MENU"
    dotted = (OFFSETT+len(name))*'-'
    options = ["[Play Pick {}]".format(NUMBER_OF_PICKS),
               "[Exit]"]
    print('{} \n{} \n{}'.format(dotted, name, dotted))
    for i, opt in enumerate(options):
        print(i+1, opt)
    print(dotted)


# 'play_pick_n', this is the game, calls the functions that get you the,
# winningNums, userNums, and checker, does with this within 3 attempts
def play_pick_n():
    global ATTEMPTS
    winningNums = get_winning_nums()

    while ATTEMPTS < MAX_ATTEMPTS:
        ATTEMPTS += 1
        userNums = get_user_nums()
        if checker(userNums, winningNums):
            break


# 'main', calls the other functions, and builds the menu
def main():
    print (PROGRAM_INFO)
    lottery_menu()
    while True:
        choice = input("\nEnter your choice[1-2]: ")
        if choice == '1':
            string = "\n[Play Pick {}]".format(NUMBER_OF_PICKS) + "selected!"
            dotted = '\n'+ len(string) * "-"

            print(dotted,
                  string,
                  dotted)

            play_pick_n()
            break

        elif choice == '2':
            print ("Thanks for playing!\n")
            break

        print("Error! Invalid input. Press any key to continue...\n")

if __name__ == '__main__':
    main()

'''

# imports
import os
import sys

# global variables
PROGRAM_INFO = '\nWritten by Brandon Sansone\nDate: 7/05/2016\n'
OFFSET = 4  # this is just for printing dashes in the menu


def modify_emp():
    '''
    modify record data from the file
    '''
    menu_name = ' ' * int(OFFSET / 2) + "MODIFY EMPLOYEE"
    dotted = (OFFSET + len(menu_name)) * '-'
    print('{} \n{} \n{}'.format(dotted, menu_name, dotted))
    # get the employee to search for
    search_name = input('Give me an employee to modify: ')
    try:
        # opens the employees.txt file for reading and the temp.txt for writing
        with open('employees.txt') as emps_file, \
                open('temp.txt', 'w') as temp_file:
            for line in emps_file:  # searches line by line for search_name
                if not line:
                    continue
                first_name, last_name, rate, hours = line.split()
                name = "{} {}".format(first_name, last_name)
                if name == search_name:  # if there is a match
                    print('\nEmployee found!')
                    # ask user for updated data
                    emp_pay_rate = input("Enter NEW rate of pay: ")
                    emp_hours_worked = input("Enter NEW hours worked: ")
                    #  write modified info to temp.txt
                    temp_file.write(name + ' ' +
                                    emp_pay_rate + ' ' +
                                    emp_hours_worked +
                                    '\n'
                                    )
                    continue
                # write the lines into temp.txt
                temp_file.write(line)
            print('\nFile written to...')
        # remove the employees.txt file because all data is transferred to
        # temp.txt
        os.remove('employees.txt')
        # rename temp.txt to employees.txt because that is our main file and
        # data has been modified
        os.rename('temp.txt', 'employees.txt')
    except FileNotFoundError:
        print("Error: File not found.")

def print_emp():
    '''
    displays only a single employee's data chosen by the user
    formatted as: firstname, lastname, grosspay (on one line of output)
    '''
    menu_name = ' ' * int(OFFSET / 2) + "EMPLOYEE LOOKUP"
    dotted = (OFFSET + len(menu_name)) * '-'
    print('{} \n{} \n{}'.format(dotted, menu_name, dotted))
    # get user input
    search_name = input('Give me an employee to search for: ')
    try:
        # open file for reading
        with open('employees.txt') as emps_file:
            for line in emps_file:
                if not line:
                    continue
                first_name, last_name, rate, hours = line.split()
                name = "{} {}".format(first_name, last_name)
                rate, hours = map(float, (rate, hours))
                if name == search_name:  # if there is a match
                    # print the employee out
                    print('\nEmployee {} found!\n'.format(name))
                    print('FIRST NAME - LAST NAME - GROSS PAY')
                    print("{} ${:4.2f}".format(name, rate * hours))
                    print(dotted)
                    break
            else:
                # informs user employee not in employees.txt
                print('Employee {} could not be found'.format(search_name))
    except FileNotFoundError:
        print("Error: File not found.")


def print_all_emps():
    '''
    displays all employee data in format:
    firstname, lastname, grosspay (on one line of output per employee)
    '''
    menu_name = ' ' * int(OFFSET / 2) + "EMPLOYEE LIST"
    dotted = (OFFSET + len(menu_name)) * '-'
    print('{} \n{} \n{}'.format(dotted, menu_name, dotted))
    try:
        # opens employees.txt for reading
        with open('employees.txt') as emps_file:
            print('FIRST NAME - LAST NAME - GROSS PAY')
            for line in emps_file:
                # goes line by line, prints each line
                first_name, last_name, rate, hours = line.split()
                name = "{} {}".format(first_name, last_name)
                rate, hours = map(float, (rate, hours))
                print("{} ${:4.2f}".format(name, rate * hours))
            print(dotted)
    except FileNotFoundError:
        print("Error: File not found.")


def print_menu():

def run_again():
    '''
    loop that asks the user if they want to run the program again,
    exits the program if they dont want to
    '''
    while True:
        choice = input('\nWant to run the program again? [Y/N]: ')
        if choice == 'Y' or choice == 'y':
            main()
            break
        elif choice == 'N' or choice == 'n':
            print("Goodbye!\n")
            sys.exit(0)

        print("Error! Invalid input. Press any key to continue...\n")
